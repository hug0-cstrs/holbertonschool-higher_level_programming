===========================
How to use 1-my_list.py
===========================
>>> MyList = __import__('1-my_list').MyList
>>> my_list = MyList()

>>> my_list.print_sorted()
[]

>>> print(my_list)
[]

>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(1)

>>> print(my_list)
[2, 3, 1]

>>> my_list.print_sorted()
[1, 2, 3]

>>> my_list.append(-1)

>>> my_list.print_sorted()
[-1, 1, 2, 3]

>>> print(my_list)
[2, 3, 1, -1]
